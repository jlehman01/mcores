Identify Cluster Proximity to Interstates and Expressways

Joel Lehman / Renaissance Planning

Summary

This analysis calculates the number of traded cluster jobs near expressways across the sate of Florida. Further, the percent of land area of all land area in Florida for each buffered distance is calculated.

Data Sources

  • ESRI Business Analyst Online Business Points - select NAICS codes to represent the following traded clusters from US Cluster Mapping:

    • Aerospace Vehicles and Defense
    • Agricultural Inputs and Services
    • Business Services
    • Communications Equipment and Services
    • Distribution and Electronic Commerce
    • Education and Knowledge Creation
    • Hospitality and Tourism
    • Information Technology and Analytical Instruments
    • Insurance Services
    • Marketing, Design, and Publishing
    • Performing Arts
    • Printing Services
    • Transportation and Logistics
    • Water Transportation

  • FDOT Roadway Characteristics (Principal Arterial-Interstate & Principal Arterial-Expressway only) - for statewide analysis

  • FDOT Interstates - for study area analysis of I-75 and I-4

  • Florida Statewide Land Use Land Cover (to identify land area)

    • Note that some analyses may report a higher number of land acres in Florida. This analysis excludes all water, including lakes, rivers, etc (but not areas designated as wetlands). As it is based on the FL land cover dataset, this estimate should be a more accurate measures of land area.

southwest_counties <- c("Hardee", "Highlands", "De Soto", "Glades", 
                   "Charlotte", "Hendry", "Collier", "Lee", "Polk")

southwest_county_sf <- st_read("K:/Projects/Florida/Features/political/countiesFDOT.shp",

  st_transform(epsg) %>% 
  filter(CO_NAME %in% southwest_counties)


roads <- st_read("K:\\Projects\\MCORES\\Features\\Transp\\funclass\\funclass.shp", quiet = TRUE) %>% 
  st_transform(epsg) %>% 
  filter(FUNCLASS %in% c("01", "02", "11", "12"))

#I4 and I75 Corridors in Study Area Only

select_roads <- st_read("K:/Projects/MCORES/Features/Transp/fdot_interstates/interstates.shp", quiet = TRUE) %>% 
  st_transform(epsg) %>% 
  filter(ROUTE == "I     4" | ROUTE == "I    75") %>% 
  filter(COUNTY %in% southwest_counties)

roads_buffer_1 <- roads %>% 
  st_buffer(1609) %>%
  st_union() %>% 
  st_sf() %>% 
  mutate(expressway_1 = 1) %>% 
  select(expressway_1)

roads_buffer_2_5 <- roads %>%

  mutate(expressway_25 = 1) %>%

roads_buffer_5 <- roads %>%

  st_union() %>%

  mutate(expressway_5 = 1) %>% 
  select(expressway_5)

select_roads_buffer_5 <- select_roads %>% 
  st_buffer(8047) %>% 
  st_union() %>% 
  st_sf() %>% 
  mutate(expressway_5 = 1) %>% 
  select(expressway_5)

#intersect points with buffers

points_with_buffers <- st_join(points, roads_buffer_1) %>% 
  st_join(roads_buffer_2_5) %>% 
  st_join(roads_buffer_5) %>% 
  mutate(expressway_1 = replace_na(expressway_1, 0),
         expressway_25 = replace_na(expressway_25, 0),
         expressway_5 = replace_na(expressway_5, 0),
         buffer = case_when(expressway_5 == 1 & expressway_25 == 0 ~ "5 Mile Buffer",
                            expressway_25 == 1 & expressway_1 == 0 ~ "2.5 Mile Buffer",
                            expressway_1 == 1 ~ "1 Mile Buffer",
                            expressway_5 == 0 ~ "Outside Buffers"))

#calculate land area

land_area <- land %>% 
  mutate(acres_florida_land = as.numeric(st_area(.))/4047) %>% 
  st_drop_geometry() %>% 
  select(acres_florida_land)

roads_buffer_1_intersect <- st_intersection(land, roads_buffer_1) %>% 
  mutate(acres_1mile = as.numeric(st_area(.))/4047) %>% 
  st_drop_geometry() %>% 
  select(acres_1mile) %>% 
  cbind(land_area) %>% 
  mutate(percent = paste0(round((acres_1mile/acres_florida_land )*100,2),"%"))
  
roads_buffer_2_5_intersect <- st_intersection(land, roads_buffer_2_5) %>% 
  mutate(acres_2.5mile = as.numeric(st_area(.))/4047) %>% 
  st_drop_geometry() %>% 
  select(acres_2.5mile) %>% 
  cbind(land_area) %>% 
  mutate(percent = paste0(round((acres_2.5mile/acres_florida_land )*100,2),"%"))
  
roads_buffer_5_intersect <- st_intersection(land, roads_buffer_5) %>% 
  mutate(acres_5mile = as.numeric(st_area(.))/4047) %>% 
  st_drop_geometry() %>% 
  select(acres_5mile) %>% 
  cbind(land_area) %>% 
  mutate(percent = paste0(round((acres_5mile/acres_florida_land )*100,2),"%"))
